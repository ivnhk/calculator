e ::= x | i | b | e1 bop e2
    | if e1 then e2 else e3
    | let x = e1 in e2

bop ::= + | * | <=

x ::= <identifiers>

i ::= <integers>

b ::= true | false

v ::= i | b
v -/->

e1 bop e2 --> e1' bop e2
    if e1 --> e1'

v1 bop e2 --> v1 bop e2'
  if e2 --> e2'

v1 bop v2 --> v
  if v is the result of primitive operation v1 bop v2
